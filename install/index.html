Nix Flakes

Run directly: nix run github:kast-lang/kast Enter temp shell with kast installed: nix shell github:kast-lang/kast

An example flake providing a devShell with kast installed (for nix develop):

{
  inputs = {
    nixpkgs.url = "nixpkgs";
    kast.url = "github:kast-lang/kast";
  };
  outputs = { kast, nixpkgs, ... }:
    let
      system = "x86_64-linux";
      pkgs = import nixpkgs { inherit system; };
    in
    {
      devShells.${system} = {
        default = pkgs.mkShell {
          packages = [
            kast.packages.${system}.kast
          ];
        };
      };
    };
}
